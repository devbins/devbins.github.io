前言

之前有介绍过使用宏来检测 C/C++ 中内存泄漏的情况，这种方案有一个缺点是只能检测自己写的代码，不能检测第三方库。

并且所有用到内存分配和释放的地方都需要修改代码，修改的成本还挺高的，并且容易漏掉。

今天就来介绍一种比较彻底的方案，使用 hook 来检测内存的分配和释放。

原理

数据结构

在进行 hook 之前还是需要把数据结构定义一下

由于我们使用 hook 的方式，就不能直接获取到调用的文件和行数了。但是我们可以获取到调用的地址，然后通过 addr2line 来还原出文件和行数。

所以最终我们的数据结构如下

1
2
3
4
5

typedef struct {
    void *p; // 内存分配的地址
    int size; // 分配的大小
    void *caller; // 调用的地址
} Mem;

hook malloc/free

定义好数据结构之后我们就需要来 hook malloc 和 free 了。

hook 的一个过程就是要找到 glib 中的 malloc 和 free 函数地址也就是函数指针。然后编写我们自己的 malloc 和 free 函数，在其中做一些记录，然后调用原来的 malloc 和 free 函数。

查找 malloc 和 free 需要使用到 dlsym 的函数，它是用来获取 symbol 的地址即函数地址，这样我们就可以通过函数指针来调用了，在这里我们要获取的 symbol 就是 malloc 和 free 。

先来看下 dlsym 的声明

1
2
3

#include <dlfcn.h>

void* dlsym(void* handle, const char* symbol);

这里有两个参数

  1. handle 表示要从这个句柄中查找 symbol 可以是 dlopen 返回的，默认也有两个伪句柄 RTLD_DEFAULT 和 RTLD_NEXT ，我们稍后再解释
  2. symbol 这个就是我们要查找的 malloc 或者 free

来看下默认的两个伪句柄

  • RTLD_DEFAULT 使用默认的共享目标搜索顺序来找到要找的符号第一次出现的地方。
  • RTLD_NEXT 在当前搜索顺序中找，在当前的目标之后，要找的符号下一次出现的地方。这个就是我们要用的伪句柄

来看下用它来查找 malloc 函数

1
2
3
4
5
6

#define _GNU_SOURCE             /* RTLD_NEXT 需要使用 */
#include <dlfcn.h>              /* dlsym 所在的头文件 */

typedef void *(*malloc_t)(size_t size);

malloc_t malloc_fun = NULL;
malloc_fun = (malloc_t)dlsym(RTLD_NEXT, "malloc");

我们先定义了一个函数指针 malloc_t ，它的声明和 malloc 是一样的，然后通过 dlsym 找到 malloc 的地址并保存下来。

接下来就需要自定义一个 malloc 函数用来替换默认的

int malloc_flag = 1;
void *malloc(size_t size) {
    if (malloc_flag) {
        malloc_flag = 0;
        void *p = malloc_fun(size);
        if (p != NULL) {
            void *caller = __builtin_return_address(0);
            for (int i = 0; i < 100; ++i) {
                if (mem[i].p == NULL) {
                    mem[i].p = p;
                    mem[i].size = size;
                    mem[i].caller = caller;
                    break;
                }
            }
        }
        malloc_flag = 1;
        return p;
    } else {
        return malloc_fun(size);
    }
}

在自定义的 malloc 中我们使用了 malloc_flag 的变量用来防止 malloc 中又调用了 malloc 导致递归调用引起的段错误，比如：在 print 中就会有 malloc 的调用。

由于需要保持 malloc 的函数签名一致，所以不能直接获取到调用的文件名和行数。但是编译器提供了 __builtin_return_address 这个函数可以获取到调用函数所在的地址，所以我们可以通过这个函数来确定代码泄漏的地方。

__builtin_return_address 这个函数有一个参数，它的作用是用来确定调用的层级，在这里我们只需要知道直接调用的地方，而不关心调用这个函数又是在哪里调用的，所以传 0 就行了。

查看内存泄漏

代码写完了，先编译执行，执行完之后，会得到一个泄漏的地址

但是执行之后发现并不能显示内存泄漏所在的文件以及对应的函数和行数，找个半天也没找到答案，最后在手写内存泄漏检测组件 - 掘金中找到办法，需要对调用的地址做一下转换，如下所示

#include <link.h>
void *convertToVma(void *p) {
  Dl_info info;
  struct link_map *link_map;
  dladdr1(p, &info, (void **)&link_map, RTLD_DL_LINKMAP);
  return p - link_map->l_addr;
}

然后再保存 caller 的时候进行一下转换

void *malloc(size_t size) {

  mem[i].p = p;
  mem[i].size = size;
  mem[i].caller = convertToVma(caller); /* 把caller 转换成 addr2line 可以识别的地址 */
}

#define _GNU_SOURCE
#include <dlfcn.h>
#include <stdio.h>
#include <stdlib.h>
#include <link.h>

typedef struct {
  void *p;
  int size;
  void *caller;
} Mem;

typedef void *(*malloc_t)(size_t size);
typedef void (*free_t)(void *p);

malloc_t malloc_fun = NULL;
free_t free_fun = NULL;
int malloc_flag = 1;
int free_flag = 1;

static Mem mem[100];

void* convertToVma(void* p){
    Dl_info info;
    struct link_map *link_map;
    dladdr1(p, &info, (void**)&link_map, RTLD_DL_LINKMAP);
    return p - link_map->l_addr;
}

void *malloc(size_t size) {
  if (malloc_flag) {
    malloc_flag = 0;
    void *p = malloc_fun(size);
    if (p != NULL) {
      void *caller = __builtin_return_address(0);
      for (int i = 0; i < 100; ++i) {
        if (mem[i].p == NULL) {
          mem[i].p = p;
          mem[i].size = size;
          mem[i].caller = convertToVma(caller);
          break;
        }
      }
    }
    malloc_flag = 1;
    return p;
  } else {
    return malloc_fun(size);
  }
}

void free(void *p) {
  if (free_flag) {
    free_flag = 0;
    if (p != NULL) {
      for (int i = 0; i < 100; ++i) {
        if (mem[i].p == p) {
          mem[i].p = NULL;
          mem[i].size = 0;
          mem[i].caller = NULL;
          break;
        }
      }
    }
    free_flag = 1;
  } else {
    free_fun(p);
  }
}

void printLeak() {
  for (int i = 0; i < 100; ++i) {
    if (mem[i].p != NULL) {
      printf("%p:%p:%d\n", mem[i].caller, mem[i].p, mem[i].size);
    }
  }
}

void hook() {
  if (malloc_fun == NULL)
    malloc_fun = (malloc_t)dlsym(RTLD_NEXT, "malloc");
  if (free_fun == NULL)
    free_fun = (free_t)dlsym(RTLD_NEXT, "free");
}

int main(int argc, char *argv[]) {
  hook();
  int *p1 = (int *)malloc(sizeof(int));
  int *p2 = (int *)malloc(sizeof(int));

  free(p1);
  printLeak();
  return 0;
}
